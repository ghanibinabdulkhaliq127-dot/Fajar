<script>
// ---------- Fireworks on canvas ----------
const canvas = document.getElementById('sky');
const ctx = canvas.getContext('2d');
function resize(){ canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
resize();
window.addEventListener('resize', resize);

let particles = [];

function launchFirework(x, y){
  const colors = ['#c9a875', '#f2eee6', '#8a7350', '#e8d5a8'];
  const count = 40;
  for(let i=0;i<count;i++){
    const angle = (Math.PI*2*i)/count;
    const speed = Math.random()*3 + 1.5;
    particles.push({
      x, y,
      vx: Math.cos(angle)*speed,
      vy: Math.sin(angle)*speed,
      life: 1,
      color: colors[Math.floor(Math.random()*colors.length)]
    });
  }
}

function animate(){
  ctx.clearRect(0,0,canvas.width, canvas.height);
  particles.forEach(p=>{
    p.x += p.vx;
    p.y += p.vy;
    p.vy += 0.02;
    p.life -= 0.012;
    if(p.life > 0){
      ctx.globalAlpha = p.life;
      ctx.fillStyle = p.color;
      ctx.beginPath();
      ctx.arc(p.x, p.y, 2, 0, Math.PI*2);
      ctx.fill();
    }
  });
  ctx.globalAlpha = 1;
  particles = particles.filter(p=>p.life > 0);
  requestAnimationFrame(animate);
}
animate();

function randomFireworkBurst(times){
  let i = 0;
  const interval = setInterval(()=>{
    const x = Math.random()*canvas.width*0.8 + canvas.width*0.1;
    const y = Math.random()*canvas.height*0.5 + canvas.height*0.1;
    launchFirework(x,y);
    i++;
    if(i>=times) clearInterval(interval);
  }, 350);
}

// ---------- Reveal sequence ----------
const sections = ['message','photo','poetry','gift','finale'];
const music = document.getElementById('bgMusic');
const soundToggle = document.getElementById('soundToggle');
let soundOn = true;

document.getElementById('openBtn').addEventListener('click', ()=>{
  document.getElementById('landing').style.display = 'none';
  soundToggle.style.display = 'block';

  music.play().catch(()=>{ /* autoplay might be blocked until interaction, this click counts as one */ });

  randomFireworkBurst(10);

  let delay = 0;
  sections.forEach((id, idx)=>{
    setTimeout(()=>{
      const el = document.getElementById(id);
      el.classList.remove('hidden-section');
      el.style.display = 'flex';
      el.querySelectorAll('.line, .urdu, .signature').forEach((line, i)=>{
        setTimeout(()=>{ line.classList.add('fade-in'); }, i*400);
      });
      if(id === 'finale'){
        randomFireworkBurst(15);
      }
      el.scrollIntoView({behavior:'smooth'});
    }, delay);
    delay += 3800;
  });
});

soundToggle.addEventListener('click', ()=>{
  soundOn = !soundOn;
  if(soundOn){ music.play(); soundToggle.textContent = 'Sound: On'; }
  else { music.pause(); soundToggle.textContent = 'Sound: Off'; }
});
</script>
